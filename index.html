class Student {
  constructor(name, age) {
    this.name = name;
    this.age = age;  
    this.balls = []; 
  }

  add(ball) {
    this.balls.push(ball);
  }

  sum() {
    const sum = this.balls.reduce((acc, ball) => acc + ball, 0);
    const average = sum / this.balls.length;
    console.log(`O'rtacha baho: ${average.toFixed(2)}`);
  }
}

const student = new Student("Ali", 20);

student.add(90);
student.add(80);
student.add(85);
student.sum();

Let's review a little Array Methods.

push()

Ma'nosi: Arrayga oxiridan element qo'shadi.

pop()

Ma'nosi: Arrayning oxiridan elementni olib tashlaydi va o'sha elementni qaytaradi.

shift()

Ma'nosi: Arrayning boshidan elementni olib tashlaydi va o'sha elementni qaytaradi.

unshift()

Ma'nosi: Arrayning boshiga element qo'shadi.

concat()

Ma'nosi: Ikki yoki undan ortiq arrayni birlashtiradi.

slice()

Ma'nosi: Arrayning bir qismini kesib oladi. Asl arrayni o'zgartirmaydi.

splice()

Ma'nosi: Arraydan elementlarni olib tashlash yoki yangi elementlar qo'shish.

indexOf()

Ma'nosi: Arrayda ma'lum bir elementning indeksini qaytaradi.

includes()

Ma'nosi: Arrayda ma'lum bir element bor-yo'qligini tekshiradi.

reverse()

Ma'nosi:Arrayni teskari tartibda qaytaradi.

join()

Ma'nosi: Array elementlarini string sifatida birlashtiradi, istalgan ajratuvchi bilan.

map()

Ma'nosi: Arrayning har bir elementi ustida funksiya bajaradi va yangi array qaytaradi.

filter()

Ma'nosi: Arraydan shartga mos keluvchi elementlarni qaytaradi.

reduce()

Ma'nosi: Array elementlarini yagona qiymatga qisqartiradi.

forEach()

Ma'nosi:Arrayning har bir elementi uchun funksiya bajaradi. Hech qanday qiymat qaytarmaydi.
